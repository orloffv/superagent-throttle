_isRateBound

returns true if throttle is bound by rate

 
  _isRateBound () {
    let throttle = this
    return (
      ((Date.now() - throttle._requestTimes[0]) < throttle.ratePer) &&
      (throttle._buffer.length > 0)
    )
  }
